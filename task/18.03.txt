2. לבדוק עבור קטע זה מה בדיוק בודק במשתנה result כלומר מה הוא מכיל.

 results = model(frame)  # הרצת YOLO על הפריים הנוכחי
        people_count = 0  # ספירת האנשים בפריים
        for result in results:  # ריצה על אובייקטים בתמונה
            for box in result.boxes:  # ריצה על כל תיבה שמייצגת אובייקט שזוהה
                cls = int(box.cls[0])  # מזהה את הקלאס של האובייקט
                if cls == 0:  # YOLO מזהה "person" כ- class 0
                    people_count += 1  # ספירת האנשים שזוהו
תשובה:
 YOLO מחזיר רשימה של תוצאות זיהוי → results
- results = [
    { "boxes": [...], "labels": ["person", "person", "car"] }
]

🔹 בתוך כל תוצאה יש רשימה של תיבות (Bounding Boxes) → result.boxes
result.boxes = [
    { "cls": [0], "coords": [100, 50, 200, 300] },  # אדם ראשון
    { "cls": [0], "coords": [250, 60, 350, 310] },  # אדם שני
    { "cls": [2], "coords": [400, 100, 550, 320] }  # רכב
🔹 אנחנו עוברים על כל תיבה בנפרד, בודקים איזה אובייקט היא מסמנת, ומזהים אם זה אדם.
